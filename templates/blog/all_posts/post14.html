좌표와 벡터의 각종 계산

// 좌표 구조체
struct Point
{
    float x, y;
};

// 벡터 구조체
struct Vector
{
    flaot x, y;
};

// (좌표) + (벡터)
Point Add_Point_Vector( Point p, Vector v )
{
    Point r;
    r.x = p.x + v.x;
    r.y = p.y + v.y;
    return ( r );
}

// (좌표) - (벡터)
Point Sub_Point_Vector( Point p, Vector v )
{
    Point r;
    r.x = p.x - v.x;
    r.y = p.y - v.y;
    return ( r );
}

// (좌표) - (좌표)
Point Sub_Point_Point( Point p0, Pointer p1 )
{
    Vector r;
    r.x = p0.x - p1.x;
    r.y = p0.y - p1.y;
    return ( r );
}

// (벡터) + (벡터)
Point Add_Vector_Vector( Vector v0, Vector v1 )
{
    Vector r;
    r.x = v0.x + v1.x
    r.y = v0.y + v1.y
    return ( r );
}

// (벡터) - (벡터)
Point Sub_Vector_Vector( Vector v0, Vector v1 )
{
    Vector r;
    r.x = v0.x - v1.x
    r.y = v0.y - v1.y
    return ( r );
}

// (좌표) x (스칼라)
Point Mul_Point_Scalar( Point p, float s )
{
    Point r;
    r.x = p.x * s;
    r.y = p.y * s;
    return ( r );
}

// (벡터) x (스칼라)
Point Mul_Vector_Scalar( Vector v, float s )
{
    Vector r;
    r.x = v.x * s;
    r.y = v.y * s;
    return ( r );
}

// (벡터) / (스칼라)
Point Div_Vector_Scalar( Vector v, float s )
{
    Vector r;
    r.x = v.x / s;
    r.y = v.y / s;
    return ( r );
}

벡터의 길이와 정규화 함수

// 벡터 길이를 계산
float Vector_Length( Vector v )
{
    float l = sqrt( v.x * v.x + v.y * v.y);
    return ( l );
}

// 벡터 정규화
Vector Vector_Normalize( Vector v )
{
    float l = Vector_Length( v );
    Vector n;
    n.x = v.x / l;
    n.y = v.y / l;
    return ( n );
}
